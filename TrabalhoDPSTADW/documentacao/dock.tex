        // Controlar a seleção de gêneros (máximo de 2)
        const checkboxes = document.querySelectorAll('input[name="generos[]"]');
        const limit = 2;
        
        checkboxes.forEach(checkbox => {
            checkbox.addEventListener('change', () => {
                const checkedCount = document.querySelectorAll('input[name="generos[]"]:checked').length;
                
                if (checkedCount > limit) {
                    checkbox.checked = false;
                    alert(`Você pode selecionar no máximo ${limit} gêneros.`);
                }
                
                // Atualizar a aparência visual
                checkboxes.forEach(cb => {
                    const label = cb.closest('.genero-option');
                    if (cb.checked) {
                        label.classList.add('selected');
                    } else {
                        label.classList.remove('selected');
                    }
                });
            });
        });
        
        // Validação do formulário
        const form = document.getElementById('jogoForm');
        form.addEventListener('submit', function(e) {
            const checkedCount = document.querySelectorAll('input[name="generos[]"]:checked').length;
            const nome = document.getElementById('nome').value;

            } else if (checkedCount > limit) {

// salvarJogo.php
function salvarJogo($conexao, $nome, $descricao, $desenvolvedor, $data_lancamento, $imagem, $idgenero) {
    $sql = "INSERT INTO jogo (nome, descricao, desenvolvedor, data_lanca, img) VALUES (?,?,?,?,?)";
    $comando = mysqli_prepare($conexao, $sql);

    mysqli_stmt_bind_param($comando, 'sssss', $nome, $descricao, $desenvolvedor, $data_lancamento, $imagem);
    mysqli_stmt_execute($comando);

    $idjogo = mysqli_insert_id($conexao);

    $sql2 = "INSERT INTO genero_jogo (genero_idgenero, jogo_idjogo) VALUES (?,?)";
    $comando2 = mysqli_prepare($conexao, $sql2);

    mysqli_stmt_bind_param($comando2, 'ii', $idgenero, $idjogo);
    $funcionou = mysqli_stmt_execute($comando2);

    mysqli_stmt_close($comando);
    mysqli_stmt_close($comando2);

    return $funcionou;
}

// Conectar ao banco de dados
$conexao = mysqli_connect("localhost", "usuario", "senha", "friv");

if (!$conexao) {
    die("Erro na conexão: " . mysqli_connect_error());
}

// Processar o formulário quando enviado
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    // Validar e sanitizar os dados
    $nome = mysqli_real_escape_string($conexao, $_POST['nome']);
    $descricao = mysqli_real_escape_string($conexao, $_POST['descricao']);
    $desenvolvedor = mysqli_real_escape_string($conexao, $_POST['desenvolvedor']);
    $data_lancamento = mysqli_real_escape_string($conexao, $_POST['data_lanca']);
    
    // Processar o upload da imagem
    $imagem_nome = '';
    if (isset($_FILES['img']) && $_FILES['img']['error'] === UPLOAD_ERR_OK) {
        $imagem_nome = basename($_FILES['img']['name']);
        $imagem_temp = $_FILES['img']['tmp_name'];
        $pasta_destino = '../uploads/' . $imagem_nome;
        
        // Mover o arquivo para o diretório de uploads
        if (!move_uploaded_file($imagem_temp, $pasta_destino)) {
            header('Location: cadastroJogo.php?status=error');
            exit();
        }
    }
    
    // Processar os gêneros selecionados
    if (isset($_POST['generos']) && is_array($_POST['generos'])) {
        $generos = $_POST['generos'];
        
        // Para cada gênero selecionado, inserir no banco
        foreach ($generos as $idgenero) {
            $sucesso = salvarJogo($conexao, $nome, $descricao, $desenvolvedor, $data_lancamento, $imagem_nome, $idgenero);
            
            if (!$sucesso) {
                header('Location: cadastroJogo.php?status=error');
                exit();
            }
        }
        
        header('Location: cadastroJogo.php?status=success');
    } else {
        header('Location: cadastroJogo.php?status=error');
    }
}

mysqli_close($conexao);
